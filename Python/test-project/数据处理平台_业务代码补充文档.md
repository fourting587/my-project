"""
Elasticsearch 客户端封装
提供索引管理、文档操作、搜索等功能
"""
from typing import List, Dict, Any, Optional
from elasticsearch import AsyncElasticsearch

from config import settings


class ESClient:
    """ES 客户端类"""
    
    def __init__(self):
        """初始化 ES 连接"""
        self.client = AsyncElasticsearch([settings.ES_URL])
    
    async def close(self):
        """关闭连接"""
        await self.client.close()
    
    async def create_index(
        self,
        index_name: str,
        mappings: Dict[str, Any],
        settings: Optional[Dict] = None
    ) -> bool:
        """
        创建索引
        
        Args:
            index_name: 索引名称
            mappings: 字段映射配置
            settings: 索引设置（分片数等）
        """
        body = {"mappings": mappings}
        if settings:
            body["settings"] = settings
        
        try:
            await self.client.indices.create(index=index_name, body=body)
            return True
        except Exception as e:
            print(f"创建索引失败: {e}")
            return False
    
    async def delete_index(self, index_name: str) -> bool:
        """删除索引"""
        try:
            await self.client.indices.delete(index=index_name)
            return True
        except Exception as e:
            print(f"删除索引失败: {e}")
            return False
    
    async def index_document(
        self,
        index_name: str,
        document: Dict[str, Any],
        doc_id: Optional[str] = None
    ) -> bool:
        """
        索引单个文档
        
        Args:
            index_name: 索引名称
            document: 文档内容
            doc_id: 指定文档ID（可选）
        """
        try:
            await self.client.index(
                index=index_name,
                id=doc_id,
                document=document
            )
            return True
        except Exception as e:
            print(f"索引文档失败: {e}")
            return False
    
    async def bulk_index(
        self,
        index_name: str,
        documents: List[Dict[str, Any]],
        id_field: Optional[str] = None
    ) -> Dict[str, int]:
        """
        批量索引文档
        
        Args:
            index_name: 索引名称
            documents: 文档列表
            id_field: 作为文档ID的字段名
        """
        from elasticsearch.helpers import async_bulk
        
        def generate_actions():
            for doc in documents:
                action = {
                    "_index": index_name,
                    "_source": doc
                }
                if id_field and id_field in doc:
                    action["_id"] = doc[id_field]
                yield action
        
        try:
            success, errors = await async_bulk(
                self.client,
                generate_actions(),
                raise_on_error=False
            )
            return {"success": success, "failed": len(errors) if errors else 0}
        except Exception as e:
            print(f"批量索引失败: {e}")
            return {"success": 0, "failed": len(documents)}
    
    async def search(
        self,
        index_name: str,
        query: str,
        fields: List[str],
        filters: Optional[Dict] = None,
        highlight: Optional[Dict] = None,
        page: int = 1,
        page_size: int = 20
    ) -> Dict[str, Any]:
        """
        执行搜索
        
        Args:
            index_name: 索引名称
            query: 搜索关键词
            fields: 搜索字段列表
            filters: 过滤条件
            highlight: 高亮配置
            page: 页码
            page_size: 每页数量
        """
        # 构建查询
        search_body = {
            "query": {
                "multi_match": {
                    "query": query,
                    "fields": fields,
                    "type": "best_fields",
                    "operator": "and"
                }
            },
            "from": (page - 1) * page_size,
            "size": page_size,
            "track_total_hits": True
        }
        
        # 添加过滤条件
        if filters:
            filter_clauses = []
            for field, value in filters.items():
                if isinstance(value, dict):
                    range_clause = {}
                    if "gte" in value:
                        range_clause["gte"] = value["gte"]
                    if "lte" in value:
                        range_clause["lte"] = value["lte"]
                    filter_clauses.append({"range": {field: range_clause}})
                else:
                    filter_clauses.append({"term": {field: value}})
            
            search_body["query"] = {
                "bool": {
                    "must": search_body["query"],
                    "filter": filter_clauses
                }
            }
        
        # 添加高亮
        if highlight:
            search_body["highlight"] = {
                "fields": {field: {} for field in highlight.get("fields", [])},
                "pre_tags": highlight.get("pre_tags", ["<mark>"]),
                "post_tags": highlight.get("post_tags", ["</mark>"])
            }
        
        # 添加聚合
        search_body["aggs"] = {
            "categories": {
                "terms": {"field": "category", "size": 10}
            }
        }
        
        try:
            response = await self.client.search(
                index=index_name,
                body=search_body
            )
            
            hits = response["hits"]["hits"]
            total = response["hits"]["total"]["value"]
            
            items = []
            for hit in hits:
                item = {
                    "_id": hit["_id"],
                    "_score": hit["_score"],
                    "_source": hit["_source"]
                }
                if "highlight" in hit:
                    item["highlight"] = hit["highlight"]
                items.append(item)
            
            return {
                "total": total,
                "page": page,
                "page_size": page_size,
                "took_ms": response["took"],
                "items": items,
                "aggregations": {
                    "categories": [
                        {"key": bucket["key"], "count": bucket["doc_count"]}
                        for bucket in response.get("aggregations", {}).get("categories", {}).get("buckets", [])
                    ]
                }
            }
            
        except Exception as e:
            print(f"搜索失败: {e}")
            return {"total": 0, "items": [], "took_ms": 0}
    
    async def suggest(
        self,
        index_name: str,
        prefix: str,
        field: str = "suggest",
        size: int = 10
    ) -> List[Dict]:
        """
        自动补全建议
        
        Args:
            index_name: 索引名称
            prefix: 输入前缀
            field: 建议字段
            size: 返回数量
        """
        body = {
            "suggest": {
                "text": prefix,
                "completion": {
                    "field": field,
                    "size": size
                }
            }
        }
        
        try:
            response = await self.client.search(index=index_name, body=body)
            suggestions = response["suggest"]["completion"][0]["options"]
            
            return [
                {"text": s["text"], "score": s["_score"]}
                for s in suggestions
            ]
        except Exception as e:
            print(f"获取建议失败: {e}")
            return []
    
    async def delete_document(self, index_name: str, doc_id: str) -> bool:
        """删除文档"""
        try:
            await self.client.delete(index=index_name, id=doc_id)
            return True
        except Exception as e:
            print(f"删除文档失败: {e}")
            return False
EOF

"""
数据源管理服务
管理数据库连接、测试连接等功能
"""
from typing import Optional, List, Dict, Any
from datetime import datetime

import aiomysql
import aiohttp
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession

from models.datasource import Datasource


class DatasourceService:
    """数据源服务类"""
    
    def __init__(self, db: AsyncSession):
        self.db = db
    
    async def get_list(
        self,
        page: int = 1,
        page_size: int = 20,
        datasource_type: Optional[str] = None,
        status: Optional[int] = None
    ) -> Dict[str, Any]:
        """
        获取数据源列表
        
        Args:
            page: 页码
            page_size: 每页数量
            datasource_type: 类型筛选
            status: 状态筛选
        """
        query = select(Datasource).where(Datasource.is_deleted == 0)
        
        if datasource_type:
            query = query.where(Datasource.type == datasource_type)
        if status is not None:
            query = query.where(Datasource.status == status)
        
        # 统计总数
        count_query = select(func.count()).select_from(query.subquery())
        total = await self.db.scalar(count_query)
        
        # 分页查询
        query = query.order_by(Datasource.created_at.desc())
        query = query.offset((page - 1) * page_size).limit(page_size)
        
        result = await self.db.execute(query)
        items = result.scalars().all()
        
        return {
            "total": total,
            "page": page,
            "page_size": page_size,
            "items": [item.to_dict() for item in items]
        }
    
    async def get_by_id(self, datasource_id: int) -> Optional[Datasource]:
        """根据ID获取数据源"""
        result = await self.db.execute(
            select(Datasource).where(
                Datasource.id == datasource_id,
                Datasource.is_deleted == 0
            )
        )
        return result.scalar_one_or_none()
    
    async def create(self, data: Dict[str, Any]) -> Datasource:
        """创建数据源"""
        # 移除密码等敏感字段的打印日志
        ds = Datasource(**data)
        self.db.add(ds)
        await self.db.flush()
        await self.db.refresh(ds)
        return ds
    
    async def update(self, datasource_id: int, data: Dict[str, Any]) -> Optional[Datasource]:
        """更新数据源"""
        ds = await self.get_by_id(datasource_id)
        if not ds:
            return None
        
        for key, value in data.items():
            if hasattr(ds, key):
                setattr(ds, key, value)
        
        await self.db.flush()
        await self.db.refresh(ds)
        return ds
    
    async def delete(self, datasource_id: int) -> bool:
        """删除数据源（软删除）"""
        ds = await self.get_by_id(datasource_id)
        if not ds:
            return False
        
        ds.is_deleted = 1
        await self.db.flush()
        return True
    
    async def test_connection(self, datasource_id: int) -> Dict[str, Any]:
        """
        测试数据源连接
        
        Returns:
            {
                "connected": bool,
                "latency_ms": float,
                "version": str or None,
                "error": str or None
            }
        """
        ds = await self.get_by_id(datasource_id)
        if not ds:
            return {"connected": False, "error": "数据源不存在"}
        
        start_time = datetime.now()
        
        try:
            if ds.type == "mysql":
                return await self._test_mysql(ds, start_time)
            elif ds.type == "api":
                return await self._test_api(ds, start_time)
            else:
                return {"connected": False, "error": f"不支持的类型: {ds.type}"}
        
        except Exception as e:
            return {
                "connected": False,
                "latency_ms": (datetime.now() - start_time).total_seconds() * 1000,
                "error": str(e)
            }
    
    async def _test_mysql(self, ds: Datasource, start_time: datetime) -> Dict[str, Any]:
        """测试 MySQL 连接"""
        conn = await aiomysql.connect(
            host=ds.host,
            port=ds.port or 3306,
            user=ds.username,
            password=ds.password,
            db=ds.database_name,
            connect_timeout=5
        )
        
        try:
            async with conn.cursor() as cur:
                await cur.execute("SELECT VERSION()")
                version = await cur.fetchone()
            
            return {
                "connected": True,
                "latency_ms": (datetime.now() - start_time).total_seconds() * 1000,
                "version": version[0] if version else None
            }
        finally:
            conn.close()
    
    async def _test_api(self, ds: Datasource, start_time: datetime) -> Dict[str, Any]:
        """测试 API 连接"""
        # 这里简化处理，实际应该根据配置发送请求
        return {
            "connected": True,
            "latency_ms": (datetime.now() - start_time).total_seconds() * 1000,
            "version": None
        }
EOF

"""
数据处理服务
提供数据清洗、转换、聚合等功能
"""
import re
from typing import List, Dict, Any, Optional

import pandas as pd
from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession


class DataCleanService:
    """数据清洗服务"""
    
    def __init__(self, db: AsyncSession):
        self.db = db
        # 定义清洗规则映射
        self.rules = {
            "trim": self._rule_trim,
            "email_format": self._rule_email_format,
            "phone_format": self._rule_phone_format,
            "range": self._rule_range,
            "unique": self._rule_unique,
            "regex": self._rule_regex,
            "map": self._rule_map,
        }
    
    async def clean_data(
        self,
        input_table: str,
        output_table: str,
        rules: List[Dict[str, Any]]
    ) -> Dict[str, int]:
        """
        清洗数据
        
        Args:
            input_table: 输入表名
            output_table: 输出表名
            rules: 清洗规则列表
        
        Returns:
            统计信息
        """
        # 读取数据
        df = await self._read_table(input_table)
        original_count = len(df)
        
        stats = {
            "original_count": original_count,
            "removed_count": 0,
            "modified_count": 0,
            "output_count": 0
        }
        
        # 应用清洗规则
        for rule in rules:
            rule_type = rule.get("rule")
            field = rule.get("field")
            action = rule.get("action", "fix")
            
            if rule_type in self.rules:
                before_count = len(df)
                df = self.rules[rule_type](df, field, rule, action)
                after_count = len(df)
                
                if action == "remove_invalid":
                    stats["removed_count"] += before_count - after_count
                else:
                    stats["modified_count"] += before_count - after_count
        
        # 保存清洗后的数据
        await self._save_dataframe(df, output_table)
        
        stats["output_count"] = len(df)
        return stats
    
    async def _read_table(self, table_name: str) -> pd.DataFrame:
        """从数据库读取表数据"""
        result = await self.db.execute(text(f"SELECT * FROM {table_name}"))
        rows = result.fetchall()
        
        if rows:
            columns = list(result.keys())
            df = pd.DataFrame([dict(row._mapping) for row in rows], columns=columns)
        else:
            df = pd.DataFrame()
        
        return df
    
    async def _save_dataframe(self, df: pd.DataFrame, table_name: str):
        """保存 DataFrame 到数据库"""
        if df.empty:
            return
        
        # 构建批量插入 SQL
        columns = list(df.columns)
        batch_size = 500
        
        for i in range(0, len(df), batch_size):
            batch = df.iloc[i:i+batch_size]
            values_list = []
            
            for _, row in batch.iterrows():
                values = [self._escape_value(v) for v in row.values]
                values_list.append(f"({', '.join(values)})")
            
            sql = f"INSERT INTO {table_name} ({', '.join(columns)}) VALUES {', '.join(values_list)}"
            await self.db.execute(text(sql))
        
        await self.db.commit()
    
    def _escape_value(self, value: Any) -> str:
        """转义 SQL 值"""
        if value is None or pd.isna(value):
            return "NULL"
        if isinstance(value, str):
            escaped = value.replace("'", "''")
            return f"'{escaped}'"
        return str(value)
    
    # ============ 清洗规则实现 ============
    
    def _rule_trim(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """去除空白字符"""
        if field in df.columns:
            df[field] = df[field].astype(str).str.strip()
        return df
    
    def _rule_email_format(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """邮箱格式验证"""
        if field not in df.columns:
            return df
        
        email_pattern = r'^[\w\.-]+@[\w\.-]+\.\w+$'
        valid_mask = df[field].astype(str).str.match(email_pattern, na=False)
        
        if action == "remove_invalid":
            return df[valid_mask]
        elif action == "set_null":
            df.loc[~valid_mask, field] = None
        
        return df
    
    def _rule_phone_format(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """手机号格式验证"""
        if field not in df.columns:
            return df
        
        pattern = config.get("pattern", r"^1[3-9]\d{9}$")
        valid_mask = df[field].astype(str).str.match(pattern, na=False)
        
        if action == "remove_invalid":
            return df[valid_mask]
        elif action == "normalize":
            df[field] = df[field].astype(str).str.replace(r'\D', '', regex=True)
        
        return df
    
    def _rule_range(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """数值范围验证"""
        if field not in df.columns:
            return df
        
        min_val = config.get("min")
        max_val = config.get("max")
        
        numeric_series = pd.to_numeric(df[field], errors='coerce')
        valid_mask = pd.Series([True] * len(df))
        
        if min_val is not None:
            valid_mask &= numeric_series >= min_val
        if max_val is not None:
            valid_mask &= numeric_series <= max_val
        
        if action == "remove_invalid":
            return df[valid_mask]
        elif action == "set_null":
            df.loc[~valid_mask, field] = None
        elif action == "clip":
            df[field] = numeric_series.clip(min_val, max_val)
        
        return df
    
    def _rule_unique(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """去重"""
        keys = config.get("keys", [field])
        keep = config.get("keep", "first")
        
        if all(k in df.columns for k in keys):
            return df.drop_duplicates(subset=keys, keep=keep)
        
        return df
    
    def _rule_regex(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """正则表达式处理"""
        if field not in df.columns:
            return df
        
        pattern = config.get("pattern", "")
        valid_mask = df[field].astype(str).str.match(pattern, na=False)
        
        if action == "remove_invalid":
            return df[valid_mask]
        elif action == "set_null":
            df.loc[~valid_mask, field] = None
        elif action == "extract":
            df[field] = df[field].astype(str).str.extract(pattern)
        
        return df
    
    def _rule_map(self, df: pd.DataFrame, field: str, config: dict, action: str) -> pd.DataFrame:
        """值映射"""
        if field not in df.columns:
            return df
        
        mapping = config.get("mapping", {})
        df[field] = df[field].map(mapping).fillna(df[field])
        
        return df


class DataAnalyticsService:
    """数据分析服务"""
    
    def __init__(self, db: AsyncSession):
        self.db = db
    
    async def execute_sql(
        self,
        sql: str,
        params: Optional[Dict] = None,
        limit: int = 1000
    ) -> Dict[str, Any]:
        """执行 SQL 查询"""
        from datetime import datetime
        
        start_time = datetime.now()
        
        # 添加 LIMIT 限制
        if "LIMIT" not in sql.upper():
            sql = f"{sql} LIMIT {limit}"
        
        result = await self.db.execute(text(sql), params or {})
        rows = result.fetchall()
        
        execution_time = (datetime.now() - start_time).total_seconds() * 1000
        
        if rows:
            columns = list(result.keys())
            data = [dict(row._mapping) for row in rows]
        else:
            columns = []
            data = []
        
        return {
            "columns": columns,
            "rows": data,
            "total": len(data),
            "execution_time_ms": round(execution_time, 2)
        }
    
    async def statistics(
        self,
        table: str,
        fields: List[str],
        group_by: Optional[List[str]] = None,
        metrics: List[str] = None,
        filters: Optional[Dict] = None
    ) -> Dict[str, Any]:
        """统计分析"""
        # 读取数据
        query = f"SELECT * FROM {table}"
        
        if filters:
            where_clauses = []
            for field, condition in filters.items():
                if isinstance(condition, dict):
                    if "gte" in condition:
                        where_clauses.append(f"{field} >= '{condition['gte']}'")
                    if "lte" in condition:
                        where_clauses.append(f"{field} <= '{condition['lte']}'")
                else:
                    where_clauses.append(f"{field} = '{condition}'")
            
            if where_clauses:
                query += " WHERE " + " AND ".join(where_clauses)
        
        result = await self.db.execute(text(query))
        rows = result.fetchall()
        
        if not rows:
            return {"summary": {}, "groups": []}
        
        columns = list(result.keys())
        df = pd.DataFrame([dict(row._mapping) for row in rows], columns=columns)
        
        # 统计指标
        metrics = metrics or ["count", "sum", "avg", "min", "max"]
        numeric_fields = df[fields].select_dtypes(include=['number']).columns.tolist()
        
        # 汇总统计
        summary = {"total_count": len(df)}
        
        for field in numeric_fields:
            if "sum" in metrics:
                summary[f"sum_{field}"] = float(df[field].sum())
            if "avg" in metrics:
                summary[f"avg_{field}"] = float(df[field].mean())
            if "min" in metrics:
                summary[f"min_{field}"] = float(df[field].min())
            if "max" in metrics:
                summary[f"max_{field}"] = float(df[field].max())
        
        # 分组统计
        groups = []
        if group_by and all(col in df.columns for col in group_by):
            grouped = df.groupby(group_by)
            
            for name, group in grouped:
                group_stat = {}
                
                if isinstance(name, tuple):
                    for i, col in enumerate(group_by):
                        group_stat[col] = name[i]
                else:
                    group_stat[group_by[0]] = name
                
                group_stat["count"] = len(group)
                
                for field in numeric_fields:
                    if "sum" in metrics:
                        group_stat[f"sum_{field}"] = float(group[field].sum())
                    if "avg" in metrics:
                        group_stat[f"avg_{field}"] = float(group[field].mean())
                
                groups.append(group_stat)
        
        return {
            "summary": summary,
            "groups": groups
        }

class CollectProcessor:
    """采集处理器"""
    
    BATCH_SIZE = 500  # 批量处理大小
    
    def __init__(self, db: AsyncSession, task: CollectTask, instance: CollectInstance):
        self.db = db
        self.task = task
        self.instance = instance
        self.total_count = 0
        self.success_count = 0
        self.fail_count = 0
    
    async def process(self):
        """根据任务类型处理采集"""
        source_type = self.task.task_type
        
        if source_type == "sql":
            await self._process_sql()
        elif source_type == "api":
            await self._process_api()
        elif source_type == "file":
            await self._process_file()
    
    async def _process_sql(self):
        """处理 SQL 数据源采集"""
        from models.datasource import Datasource
        
        # 获取数据源配置
        result = await self.db.execute(
            select(Datasource).where(Datasource.id == self.task.datasource_id)
        )
        ds = result.scalar_one()
        
        source_config = self.task.source_config
        sql = source_config.get("sql", "")
        primary_key = source_config.get("primary_key", "id")
        
        # 替换时间变量
        if self.instance.start_time:
            sql = sql.replace(":last_sync_time", f"'{self.instance.start_time}'")
        if self.instance.end_time:
            sql = sql.replace(":end_time", f"'{self.instance.end_time}'")
        
        # 连接源数据库
        conn = await aiomysql.connect(
            host=ds.host,
            port=ds.port or 3306,
            user=ds.username,
            password=ds.password,
            db=ds.database_name
        )
        
        try:
            async with conn.cursor(aiomysql.DictCursor) as cur:
                await cur.execute(sql)
                
                batch = []
                while True:
                    rows = await cur.fetchmany(self.BATCH_SIZE)
                    if not rows:
                        break
                    
                    batch.extend(rows)
                    self.total_count += len(rows)
                    
                    # 达到批量大小时保存
                    if len(batch) >= self.BATCH_SIZE:
                        await self._save_batch(batch)
                        batch = []
                        await self._update_progress()
                
                # 保存剩余数据
                if batch:
                    await self._save_batch(batch)
                    await self._update_progress()
        
        finally:
            conn.close()
    
    async def _process_api(self):
        """处理 API 数据源采集"""
        source_config = self.task.source_config
        url = source_config.get("url", "")
        method = source_config.get("method", "GET")
        headers = source_config.get("headers", {})
        params = source_config.get("params", {})
        data_path = source_config.get("data_path", "")
        
        async with aiohttp.ClientSession() as session:
            all_data = []
            page = 1
            
            while True:
                page_params = {**params, "page": page, "limit": self.BATCH_SIZE}
                
                async with session.request(
                    method, url, headers=headers, params=page_params
                ) as resp:
                    result = await resp.json()
                    
                    # 提取数据
                    items = result
                    if data_path:
                        for key in data_path.split("."):
                            items = items.get(key, [])
                    
                    if not items:
                        break
                    
                    all_data.extend(items)
                    self.total_count += len(items)
                    
                    # 判断是否还有更多数据
                    if len(items) < self.BATCH_SIZE:
                        break
                    
                    page += 1
                    
                    # 批量保存
                    if len(all_data) >= self.BATCH_SIZE:
                        await self._save_batch(all_data[:self.BATCH_SIZE])
                        all_data = all_data[self.BATCH_SIZE:]
                        await self._update_progress()
            
            # 保存剩余数据
            if all_data:
                await self._save_batch(all_data)
                await self._update_progress()
    
    async def _process_file(self):
        """处理文件数据源（简化实现）"""

        pass
    
    async def _save_batch(self, batch: List[Dict[str, Any]]):
        """批量保存数据到目标表"""
        if not batch:
            return
        
        try:
            # 应用字段映射
            field_mapping = self.task.field_mapping or {}
            mapped_batch = []
            
            for row in batch:
                if field_mapping:
                    mapped_row = {}
                    for source_field, target_field in field_mapping.items():
                        mapped_row[target_field] = row.get(source_field)
                    mapped_batch.append(mapped_row)
                else:
                    mapped_batch.append(row)
            
            # 构建批量插入 SQL
            if mapped_batch:
                columns = list(mapped_batch[0].keys())
                placeholders = ", ".join([f":{col}" for col in columns])
                column_names = ", ".join(columns)
                
                # 构建参数列表
                params_list = []
                for item in mapped_batch:
                    params_list.append({k: v for k, v in item.items()})
                
                sql = f"INSERT INTO {self.task.target_table} ({column_names}) VALUES ({placeholders})"
                
                await self.db.execute(text(sql), params_list)
                await self.db.commit()
            
            self.success_count += len(batch)
            
        except Exception as e:
            self.fail_count += len(batch)
            print(f"保存批次失败: {e}")
    
    async def _update_progress(self):
        """更新任务进度"""
        if self.total_count > 0:
            self.instance.progress = min(100, int(
                self.success_count / self.total_count * 100
            ))
        self.instance.total_count = self.total_count
        self.instance.success_count = self.success_count
        self.instance.fail_count = self.fail_count
        await self.db.commit()

@router.post("/")
async def search(
    query: str = Query(..., description="搜索关键词"),
    index: str = Query(..., description="索引名称"),
    fields: List[str] = Query(default=["title", "content"]),
    filters: Optional[Dict[str, Any]] = None,
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    current_user: User = Depends(get_current_user)
):
    """
    执行全文搜索
    
    Args:
        query: 搜索关键词
        index: 索引名称
        fields: 搜索字段列表
        filters: 过滤条件
        page: 页码
        page_size: 每页数量
    """
    es = ESClient()
    try:
        result = await es.search(
            index_name=index,
            query=query,
            fields=fields,
            filters=filters,
            highlight={"fields": ["title", "content"]},
            page=page,
            page_size=page_size
        )
        return {"code": 200, "message": "success", "data": result}
    finally:
        await es.close()

"""
用户管理 API
提供用户的增删改查功能
"""
from typing import Optional

from fastapi import APIRouter, Depends, Query, HTTPException, status
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession

from database import get_db
from models.user import User
from routers.auth import get_current_user, get_password_hash

router = APIRouter()


@router.get("/")
async def get_users(
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    keyword: Optional[str] = Query(None),
    status: Optional[int] = Query(None),
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """获取用户列表"""
    query = select(User).where(User.is_deleted == 0)
    
    if keyword:
        query = query.where(
            (User.username.contains(keyword)) |
            (User.nickname.contains(keyword)) |
            (User.email.contains(keyword))
        )
    
    if status is not None:
        query = query.where(User.status == status)
    
    # 统计总数
    count_query = select(func.count()).select_from(query.subquery())
    total = await db.scalar(count_query)
    
    # 分页
    query = query.order_by(User.created_at.desc())
    query = query.offset((page - 1) * page_size).limit(page_size)
    
    result = await db.execute(query)
    items = result.scalars().all()
    
    return {
        "code": 200,
        "message": "success",
        "data": {
            "total": total,
            "page": page,
            "page_size": page_size,
            "items": [item.to_dict() for item in items]
        }
    }


@router.post("/")
async def create_user(
    data: dict,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """创建用户"""
    # 检查用户名是否已存在
    result = await db.execute(
        select(User).where(User.username == data.get("username"))
    )
    if result.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="用户名已存在")
    
    # 加密密码
    if "password" in data:
        data["password"] = get_password_hash(data["password"])
    
    user = User(**data)
    db.add(user)
    await db.flush()
    await db.refresh(user)
    
    return {"code": 200, "message": "创建成功", "data": user.to_dict()}


@router.get("/{user_id}")
async def get_user(
    user_id: int,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """获取用户详情"""
    result = await db.execute(
        select(User).where(User.id == user_id, User.is_deleted == 0)
    )
    user = result.scalar_one_or_none()
    
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")
    
    return {"code": 200, "message": "success", "data": user.to_dict()}


@router.put("/{user_id}")
async def update_user(
    user_id: int,
    data: dict,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """更新用户信息"""
    result = await db.execute(
        select(User).where(User.id == user_id, User.is_deleted == 0)
    )
    user = result.scalar_one_or_none()
    
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")
    
    # 更新字段
    for key, value in data.items():
        if hasattr(user, key) and key != "id":
            setattr(user, key, value)
    
    await db.flush()
    await db.refresh(user)
    
    return {"code": 200, "message": "更新成功", "data": user.to_dict()}


@router.delete("/{user_id}")
async def delete_user(
    user_id: int,
    db: AsyncSession = Depends(get_db),

    )

    user.is_deleted = 1
    await db.flush()

数据源相关模型

from sqlalchemy import Column, String, Integer, JSON, ForeignKey, BigInteger
from sqlalchemy.orm import relationship

class Datasource(BaseModel):
    """数据源配置表"""
    __tablename__ = "datasource"
    
    name = Column(String(100), nullable=False, comment="数据源名称")
    type = Column(String(20), nullable=False, comment="类型：mysql/postgresql/api/file")
    description = Column(String(500), default=None, comment="描述")

    host = Column(String(200), default=None, comment="主机地址")
    port = Column(Integer, default=None, comment="端口")
    database_name = Column(String(100), default=None, comment="数据库名")

    api_method = Column(String(10), default=None, comment="请求方法")
