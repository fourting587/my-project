# 数据处理/分析平台 - 完整工程实施文档

## 文档说明

本文档提供从零开始搭建完整数据处理平台的详细步骤，包含所有源代码和详细注释。按照文档顺序执行即可完成项目部署。

**环境配置：**
- CPU: 4核
- 内存: 8GB
- 磁盘: 160GB
- 操作系统: Ubuntu 24.04 LTS

## 第一阶段：系统环境准备

### 步骤 1.1：更新系统并安装基础工具

```bash
# 更新系统软件包
sudo apt update && sudo apt upgrade -y

# 安装基础工具
sudo apt install -y \
    vim \
    git \
    curl \
    wget \
    net-tools \
    htop \
    tree \
    unzip

# 设置时区
sudo timedatectl set-timezone Asia/Shanghai
```

### 步骤 1.2：安装 Docker 和 Docker Compose

```bash
# 安装 Docker
sudo apt install -y docker.io

# 启动 Docker 服务
sudo systemctl enable docker
sudo systemctl start docker

# 添加当前用户到 docker 组（免 sudo 运行 docker）
sudo usermod -aG docker $USER

# 安装 Docker Compose V2 插件
sudo apt install -y docker-compose-plugin

# 验证安装
docker --version
docker compose version
```

**注意：** 添加用户到 docker 组后，需要 **重新登录** 或 **重启系统** 才能生效。

### 步骤 1.3：创建项目目录结构

```bash
# 创建项目根目录
sudo mkdir -p /data-platform
sudo chown $USER:$USER /data-platform
cd /data-platform

# 创建所有必要的子目录
mkdir -p mysql/{data,init}      # MySQL 数据和初始化脚本
mkdir -p redis/data             # Redis 数据
mkdir -p es/data                # Elasticsearch 数据（重要！）
```

### 步骤 2.5：创建 Nginx 配置文件

```bash
cat > nginx/nginx.conf << 'EOF'
# Nginx 配置文件
user nginx;
worker_processes auto;              # 自动根据 CPU 核心数设置
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

events {
    worker_connections 1024;        # 单个 worker 的连接数
    use epoll;                      # 使用 epoll 事件模型（Linux 高效）
    multi_accept on;                # 同时接受多个连接
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # 日志格式定义
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for"';

    access_log /var/log/nginx/access.log main;

    # 性能优化
    sendfile on;                    # 启用 sendfile 高效传输
    tcp_nopush on;                  # 防止网络拥塞
    tcp_nodelay on;                 # 实时发送小数据包
    keepalive_timeout 65;           # 长连接超时时间
    types_hash_max_size 2048;
    client_max_body_size 100M;      # 最大上传文件大小

    # Gzip 压缩配置
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;              # 压缩级别（1-9，6 是平衡点）
    gzip_min_length 1024;           # 小于 1KB 不压缩
    gzip_types text/plain text/css application/json application/javascript 
               text/xml application/xml application/xml+rss text/javascript;

    # 限流配置（防刷）
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=auth:10m rate=5r/m;

    # 上游服务配置（FastAPI 应用）
    upstream app {
        server app:8000 max_fails=3 fail_timeout=30s;
        keepalive 32;               # 保持长连接数
    }

    # HTTP 服务器配置
    server {
        listen 80;
        server_name localhost;
        
        # 安全响应头
        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header X-XSS-Protection "1; mode=block" always;

        # 前端静态文件（Vue 构建产物）
        location / {
            root /usr/share/nginx/html;
            try_files $uri $uri/ /index.html;  # 支持 Vue Router
            expires 1d;                        # 缓存 1 天
        }

        # API 代理（转发到 FastAPI）
        location /api/ {
            limit_req zone=api burst=20 nodelay;  # 限流配置
            proxy_pass http://app/;               # 转发到 upstream
            proxy_http_version 1.1;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_set_header Connection "";
            proxy_connect_timeout 60s;
            proxy_send_timeout 60s;
            proxy_read_timeout 60s;
        }

        # WebSocket 支持（用于实时监控）
        location /ws/ {
            proxy_pass http://app/;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "upgrade";
            proxy_read_timeout 86400s;
        }

        # 健康检查端点
        location /health {
            access_log off;
            return 200 "healthy\n";
            add_header Content-Type text/plain;
        }
    }
}
EOF
```

```bash
cat > prometheus/alert_rules.yml << 'EOF'
groups:
  - name: system_alerts
    rules:
      # 内存使用率告警
      - alert: HighMemoryUsage
        expr: |
          (1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100 > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "内存使用率过高"
          description: "内存使用率超过 85%，当前值: {{ $value }}%"

      # 磁盘使用率告警
      - alert: HighDiskUsage
        expr: |
          (1 - (node_filesystem_avail_bytes / node_filesystem_size_bytes)) * 100 > 85
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "磁盘使用率过高"
          description: "磁盘使用率超过 85%"

      # 应用响应时间告警
      - alert: AppHighLatency
        expr: |
          histogram_quantile(0.95, rate(http_request_duration_seconds_bucket[5m])) > 1
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "应用响应延迟过高"
```

## 第三阶段：Docker Compose 编排配置

### 步骤 3.1：创建主 Docker Compose 文件

```bash
cd /data-platform

cat > docker-compose.yml << 'EOF'
# ============================================
# Docker Compose 配置 - 数据处理平台
# 硬件配置：4核CPU / 8GB内存 / 160GB磁盘
# ============================================
version: '3.8'

services:
  # ============================================
  # MySQL 8.0 - 关系型数据库
  # 内存限制：1GB
  # ============================================
  mysql:
    image: mysql:8.0
    container_name: mysql
    restart: unless-stopped               # 除非手动停止，否则总是重启
    environment:
      # 从 .env 文件读取环境变量
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD}
      MYSQL_DATABASE: ${MYSQL_DATABASE}
      MYSQL_USER: ${MYSQL_USER}
      MYSQL_PASSWORD: ${MYSQL_PASSWORD}
    volumes:
      # 数据持久化挂载
      - ./mysql/data:/var/lib/mysql       # MySQL 数据文件
      - ./mysql/init:/docker-entrypoint-initdb.d  # 初始化脚本
      - ./mysql/my.cnf:/etc/mysql/conf.d/my.cnf:ro  # 配置文件（只读）
      - ./backups:/backups                # 备份目录
    ports:
      - "3306:3306"                       # 映射端口
    deploy:
      resources:
        limits:
          cpus: '1.0'                      # 限制 1 核 CPU
          memory: 1G                       # 限制 1GB 内存
        reservations:
          memory: 512M                     # 预留 512MB
    networks:
      - backend
    healthcheck:                           # 健康检查
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost", "-u", "root", "-p${MYSQL_ROOT_PASSWORD}"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 30s

  # ============================================
  # Redis 7 - 缓存和消息代理
  # 内存限制：512MB
  # ============================================
  redis:
    image: redis:7-alpine
    container_name: redis
    restart: unless-stopped
    command: >                            # 启动命令覆盖
      redis-server
      --appendonly yes                    # 开启 AOF 持久化
      --maxmemory 512mb                   # 最大内存限制
      --maxmemory-policy allkeys-lru      # 内存淘汰策略
      --requirepass ${REDIS_PASSWORD}     # 密码认证
    volumes:
      - ./redis/data:/data                # 数据持久化
    ports:
      - "6379:6379"
    deploy:
      resources:
        limits:
          cpus: '0.5'
          memory: 512M
        reservations:
          memory: 128M
    networks:
```

```bash
"""
基础模型类
提供所有数据模型的公共字段和方法
"""
from datetime import datetime
from typing import Any, Dict

from sqlalchemy import Column, DateTime, BigInteger, Integer
from sqlalchemy.orm import declared_attr

from database import Base


class BaseModel(Base):
    """
    基础模型类
    所有业务模型都应继承此类
    """
    __abstract__ = True                        # 抽象类，不创建表
    
    @declared_attr.directive
    def __tablename__(cls) -> str:
        """自动将类名转为表名（小写）"""
        return cls.__name__.lower()
    
    # 公共字段：所有表都包含
    id = Column(BigInteger, primary_key=True, index=True, autoincrement=True, comment="主键ID")
    created_at = Column(DateTime, default=datetime.utcnow, nullable=False, comment="创建时间")
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, nullable=False, comment="更新时间")
    is_deleted = Column(Integer, default=0, nullable=False, comment="是否删除：0-否 1-是")
    
    def to_dict(self) -> Dict[str, Any]:
        """
        将模型对象转为字典
        用于 API 响应序列化
        """
        return {
            column.name: getattr(self, column.name)
            for column in self.__table__.columns
        }
    
    def __repr__(self) -> str:
        """字符串表示，方便调试"""
        return f"<{self.__class__.__name__}(id={self.id})>"
```

```bash
from typing import Optional

    """验证密码"""

def get_password_hash(password: str) -> str:

def create_access_token(data: dict, expires_delta: Optional[timedelta] = None):
    """
    创建 JWT 访问令牌
    
    Args:
        data: 要编码的数据
        expires_delta: 过期时间增量
    """
    to_encode = data.copy()
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)

    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)

async def get_current_user(
    token: str = Depends(oauth2_scheme),
    db: AsyncSession = Depends(get_db)

    获取当前登录用户
    通过 JWT Token 解析用户身份
    """
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="无效的认证凭据",
        headers={"WWW-Authenticate": "Bearer"},
    )
    
    try:
        # 解码 JWT
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        user_id: str = payload.get("sub")
        if user_id is None:
            raise credentials_exception
    except JWTError:
        raise credentials_exception
    
    # 查询用户
    result = await db.execute(select(User).where(User.id == int(user_id)))
    user = result.scalar_one_or_none()
    
    if user is None:
        raise credentials_exception
    
    return user
```

```bash
"""Elasticsearch 客户端封装"""
from typing import List, Dict, Any, Optional
from elasticsearch import AsyncElasticsearch

from config import settings


class ESClient:
    """ES 客户端类"""
    
    def __init__(self):
        self.client = AsyncElasticsearch([settings.ES_URL])
    
    async def close(self):
        await self.client.close()
    
    async def search(
        self,
        index_name: str,
        query: str,
        fields: List[str],
        filters: Optional[Dict] = None,
        highlight: Optional[Dict] = None,
        page: int = 1,
        page_size: int = 20
    ) -> Dict[str, Any]:
        search_body = {
            "query": {
                "multi_match": {
                    "query": query,
                    "fields": fields,
                    "type": "best_fields",
                    "operator": "and"
                }
            },
            "from": (page - 1) * page_size,
            "size": page_size,
            "track_total_hits": True
        }
        
        if filters:
            filter_clauses = []
            for field, value in filters.items():
                if isinstance(value, dict):
                    range_clause = {}
                    if "gte" in value:
                        range_clause["gte"] = value["gte"]
                    if "lte" in value:
                        range_clause["lte"] = value["lte"]
                    filter_clauses.append({"range": {field: range_clause}})
                else:
                    filter_clauses.append({"term": {field: value}})
            
            search_body["query"] = {
                "bool": {
                    "must": search_body["query"],
                    "filter": filter_clauses
                }
            }
        
        if highlight:
            search_body["highlight"] = {
                "fields": {field: {} for field in highlight.get("fields", [])},
                "pre_tags": highlight.get("pre_tags", ["<mark>"]),
                "post_tags": highlight.get("post_tags", ["</mark>"])
            }
        
        response = await self.client.search(index=index_name, body=search_body)
        
        hits = response["hits"]["hits"]
        total = response["hits"]["total"]["value"]
        
        items = []
        for hit in hits:
            item = {
                "_id": hit["_id"],
                "_score": hit["_score"],
                "_source": hit["_source"]
            }
            if "highlight" in hit:
                item["highlight"] = hit["highlight"]
            items.append(item)
        
        return {
            "total": total,
            "page": page,
            "page_size": page_size,
            "took_ms": response["took"],
            "items": items
        }
EOF
```

```bash
            action = rule.get("action", "fix")
            
            if rule_type in self.rules:
                df = self.rules[rule_type](df, field, rule, action)
        
        await self._save_dataframe(df, output_table)
        
        return {
            "original_count": original_count,
            "output_count": len(df)
        }
    
    async def _read_table(self, table_name: str) -> pd.DataFrame:
        result = await self.db.execute(text(f"SELECT * FROM {table_name}"))
        rows = result.fetchall()
        
        if rows:
            columns = list(result.keys())
            df = pd.DataFrame([dict(row._mapping) for row in rows], columns=columns)
        else:
            df = pd.DataFrame()
        
        return df
    
    async def _save_dataframe(self, df: pd.DataFrame, table_name: str):
        if df.empty:
            return
        
        columns = list(df.columns)
        batch_size = 500
        
        for i in range(0, len(df), batch_size):
            batch = df.iloc[i:i+batch_size]
            values_list = []
            
            for _, row in batch.iterrows():
                values = [self._escape_value(v) for v in row.values]
                values_list.append(f"({', '.join(values)})")
            
            sql = f"INSERT INTO {table_name} ({', '.join(columns)}) VALUES {', '.join(values_list)}"
            await self.db.execute(text(sql))
        
        await self.db.commit()
    
    def _escape_value(self, value):
        if value is None or pd.isna(value):
            return "NULL"
        if isinstance(value, str):
            escaped = value.replace("'", "''")
            return f"'{escaped}'"
        return str(value)
    
    def _rule_trim(self, df, field, config, action):
        if field in df.columns:
            df[field] = df[field].astype(str).str.strip()
        return df
    
    def _rule_email_format(self, df, field, config, action):
        if field not in df.columns:
            return df
        
        pattern = r'^[\w\.-]+@[\w\.-]+\.\w+$'
        valid = df[field].astype(str).str.match(pattern, na=False)
        
        if action == "remove_invalid":
            return df[valid]
        return df
    
    def _rule_range(self, df, field, config, action):
        if field not in df.columns:
            return df
        
        min_val = config.get("min")
        max_val = config.get("max")
        numeric = pd.to_numeric(df[field], errors='coerce')
        
        valid = pd.Series([True] * len(df))
        if min_val is not None:
            valid &= numeric >= min_val
        if max_val is not None:
            valid &= numeric <= max_val
        
        if action == "remove_invalid":
            return df[valid]
        return df


class DataAnalyticsService:
    def __init__(self, db: AsyncSession):
        self.db = db
    
    async def execute_sql(self, sql: str, params=None, limit=1000):
        from datetime import datetime
        
        start_time = datetime.now()
        
        if "LIMIT" not in sql.upper():
            sql = f"{sql} LIMIT {limit}"
        
        result = await self.db.execute(text(sql), params or {})
        rows = result.fetchall()
        
        execution_time = (datetime.now() - start_time).total_seconds() * 1000
        
        if rows:
            columns = list(result.keys())
            data = [dict(row._mapping) for row in rows]
        else:
            columns = []
            data = []
        
        return {
            "columns": columns,
            "rows": data,
            "total": len(data),
            "execution_time_ms": round(execution_time, 2)
        }
```

```bash
"""数据采集 Celery 任务"""
import asyncio
from datetime import datetime
from typing import List, Dict, Any

import aiomysql
from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession

from tasks.celery_app import celery_app
from database import AsyncSessionLocal
from models.collect import CollectInstance, CollectTask


@celery_app.task(bind=True, max_retries=3)
def execute_collect_task(self, instance_id: int):
    asyncio.run(_execute_collect_async(instance_id))


async def _execute_collect_async(instance_id: int):
    async with AsyncSessionLocal() as db:
        try:
            from sqlalchemy import select
            
            result = await db.execute(
                select(CollectInstance).where(CollectInstance.id == instance_id)
            )
            instance = result.scalar_one()
            
            instance.status = "running"
            instance.start_at = datetime.now()
            await db.commit()
            
            result = await db.execute(
                select(CollectTask).where(CollectTask.id == instance.task_id)
            )
            task = result.scalar_one()
            
            processor = CollectProcessor(db, task, instance)
            await processor.process()
            
            instance.status = "completed"
            instance.end_at = datetime.now()
            instance.duration_ms = int(
                (instance.end_at - instance.start_at).total_seconds() * 1000
            )
            await db.commit()
            
        except Exception as exc:
            instance.status = "failed"
            instance.error_msg = str(exc)[:500]
            instance.end_at = datetime.now()
            await db.commit()
            raise self.retry(exc=exc, countdown=60)


class CollectProcessor:
    BATCH_SIZE = 500
    
    def __init__(self, db: AsyncSession, task: CollectTask, instance: CollectInstance):
        self.db = db
        self.task = task
        self.instance = instance
        self.total_count = 0
        self.success_count = 0
    
    async def process(self):
        if self.task.task_type == "sql":
            await self._process_sql()
    
    async def _process_sql(self):
        from models.datasource import Datasource
        from sqlalchemy import select
        
        result = await self.db.execute(
            select(Datasource).where(Datasource.id == self.task.datasource_id)
        )
        ds = result.scalar_one()
        
        source_config = self.task.source_config
        sql = source_config.get("sql", "")
        
        if self.instance.start_time:
            sql = sql.replace(":last_sync_time", f"'{self.instance.start_time}'")
        
        conn = await aiomysql.connect(
            host=ds.host,
            port=ds.port or 3306,
            user=ds.username,
            password=ds.password,
            db=ds.database_name
        )
        
        try:
            async with conn.cursor(aiomysql.DictCursor) as cur:
                await cur.execute(sql)
                
                batch = []
                while True:
                    rows = await cur.fetchmany(self.BATCH_SIZE)
                    if not rows:
                        break
                    
                    batch.extend(rows)
                    self.total_count += len(rows)
                    
                    if len(batch) >= self.BATCH_SIZE:
                        await self._save_batch(batch)
                        batch = []
                        await self._update_progress()
                
                if batch:
                    await self._save_batch(batch)
                    await self._update_progress()
        finally:
            conn.close()
    
    async def _save_batch(self, batch: List[Dict]):
        if not batch:
            return
        
        try:
            field_mapping = self.task.field_mapping or {}
            mapped_batch = []
            
            for row in batch:
                if field_mapping:
                    mapped_row = {target: row.get(source) for source, target in field_mapping.items()}
                else:
                    mapped_row = row
                mapped_batch.append(mapped_row)
            
            if mapped_batch:
                columns = list(mapped_batch[0].keys())
                placeholders = ", ".join([f":{col}" for col in columns])
                column_names = ", ".join(columns)
                
                sql = f"INSERT INTO {self.task.target_table} ({column_names}) VALUES ({placeholders})"
                
                await self.db.execute(text(sql), mapped_batch)
                await self.db.commit()
            
            self.success_count += len(batch)
        except Exception as e:
            print(f"保存批次失败: {e}")
    
    async def _update_progress(self):
        if self.total_count > 0:
            self.instance.progress = min(100, int(self.success_count / self.total_count * 100))
        self.instance.total_count = self.total_count
        self.instance.success_count = self.success_count
        await self.db.commit()
EOF
```

```bash
    data: dict,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)

        select(User).where(User.username == data.get("username"))
    )

        data["password"] = get_password_hash(data["password"])

    db.add(user)
    await db.flush()
    await db.refresh(user)
    
    return {"code": 200, "message": "创建成功", "data": user.to_dict()}
```
